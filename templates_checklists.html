OBS WebSocket Automation

Script Python para controle automatizado do OBS Studio via WebSocket. 

import obswebsocket
from obswebsocket import obsws, requests
import time
import schedule

class OBSController:
    def __init__(self, host="localhost", port=4444, password="your_password"):
        self.ws = obsws(host, port, password)
        self.ws.connect()
    
    def start_streaming(self):
        try:
            self.ws.call(requests.StartStreaming())
            print("Live iniciada com sucesso!")
        except Exception as e:
            print(f"Erro ao iniciar live: {e}")
    
    def stop_streaming(self):
        try:
            self.ws.call(requests.StopStreaming())
            print("Live encerrada com sucesso!")
        except Exception as e:
            print(f"Erro ao encerrar live: {e}")
    
    def switch_scene(self, scene_name):
        try:
            self.ws.call(requests.SetCurrentScene(scene_name))
            print(f"Cena alterada para: {scene_name}")
        except Exception as e:
            print(f"Erro ao alterar cena: {e}")

# Uso do script
obs = OBSController()

# Agendar lives
schedule.every().day.at("09:00").do(obs.start_streaming)
schedule.every().day.at("18:00").do(obs.stop_streaming)

# Loop de execu√ß√£o
while True:
    schedule.run_pending()
    time.sleep(60)

YouTube Live API

Script para gerenciar lives automaticamente via YouTube API v3. 

from googleapiclient.discovery import build
from google.oauth2 import service_account
import datetime
import pytz

class YouTubeLiveManager:
    def __init__(self, credentials_file):
        self.credentials = service_account.Credentials.from_service_account_file(
            credentials_file,
            scopes=['https://www.googleapis.com/auth/youtube.force-ssl']
        )
        self.youtube = build('youtube', 'v3', credentials=self.credentials)
    
    def create_live_broadcast(self, title, description, scheduled_time):
        broadcast = self.youtube.liveBroadcasts().insert(
            part="snippet,status,contentDetails",
            body={
                "snippet": {
                    "title": title,
                    "description": description,
                    "scheduledStartTime": scheduled_time.isoformat(),
                    "isDefaultBroadcast": False
                },
                "status": {
                    "privacyStatus": "public",
                    "selfDeclaredMadeForKids": False
                },
                "contentDetails": {
                    "enableAutoStart": True,
                    "enableAutoStop": False,
                    "enableDvr": True,
                    "enableContentEncryption": True,
                    "recordFromStart": True,
                    "startWithSlate": False
                }
            }
        ).execute()
        
        return broadcast['id']
    
    def bind_stream_to_broadcast(self, broadcast_id, stream_id):
        bind = self.youtube.liveBroadcasts().bind(
            part="id,contentDetails",
            id=broadcast_id,
            streamId=stream_id
        ).execute()
        
        return bind
    
    def create_live_stream(self, title):
        stream = self.youtube.liveStreams().insert(
            part="snippet,cdn,contentDetails",
            body={
                "snippet": {
                    "title": title
                },
                "cdn": {
                    "format": "720p",
                    "ingestionType": "rtmp"
                },
                "contentDetails": {
                    "isReusable": True
                }
            }
        ).execute()
        
        return stream['id'], stream['cdn']['ingestionInfo']

# Uso do script
youtube = YouTubeLiveManager('credentials.json')

# Criar live stream
stream_id, ingestion_info = youtube.create_live_stream('Bitcoin Live 24/7')

# Criar broadcast
scheduled_time = datetime.datetime.now(pytz.UTC) + datetime.timedelta(minutes=10)
broadcast_id = youtube.create_live_broadcast(
    'Bitcoin AO VIVO: An√°lise 24/7',
    'Acompanhe a an√°lise t√©cnica do Bitcoin em tempo real',
    scheduled_time
)

# Vincular stream ao broadcast
youtube.bind_stream_to_broadcast(broadcast_id, stream_id)

print(f"Live criada com sucesso!")
print(f"Broadcast ID: {broadcast_id}")
print(f"Stream Key: {ingestion_info['streamName']}")
print(f"RTMP URL: {ingestion_info['ingestionAddress']}")
